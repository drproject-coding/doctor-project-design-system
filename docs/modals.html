Modals

Full-screen overlay with a centered dialog. Features a purple header, body content area, and footer with action buttons.

Markup

<div class="bru-overlay" id="demoOverlay" hidden>
  <div class="bru-modal">
    <div class="bru-modal__header">
      <h3 class="bru-modal__title">Title</h3>
      <button class="bru-modal__close">&times;</button>
    </div>
    <div class="bru-modal__body">
      <p class="bru-text">Body content here.</p>
    </div>
    <div class="bru-modal__footer">
      <button class="bru-btn bru-btn--ghost">Cancel</button>
      <button class="bru-btn bru-btn--primary">Confirm</button>
    </div>
  </div>
</div>

JavaScript

const overlay = document.getElementById('demoOverlay');

// Open
document.getElementById('openModal').addEventListener('click', () => {
  overlay.removeAttribute('hidden');
});

// Close via X button
document.getElementById('closeModal').addEventListener('click', () => {
  overlay.setAttribute('hidden', '');
});

// Close via backdrop click
overlay.addEventListener('click', (e) => {
  if (e.target === e.currentTarget) {
    e.currentTarget.setAttribute('hidden', '');
  }
});
