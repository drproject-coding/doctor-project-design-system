Usage

/* Reference any token in your CSS */
.my-element {
  color: var(--bru-purple);
  background: var(--bru-cream);
  border: 2px solid var(--bru-black);
}

/* Dark mode overrides happen automatically via tokens.css */
body.dark-mode {
  --bru-cream: #1A1A1A;
  --bru-black: #FFFFFF;
  /* ... all tokens swap */
}
