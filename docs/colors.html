Usage

/* Reference any token in your CSS */
.my-element {
  color: var(--drp-purple);
  background: var(--drp-cream);
  border: 2px solid var(--drp-black);
}

/* Dark mode overrides happen automatically via tokens.css */
body.dark-mode {
  --drp-cream: #1A1A1A;
  --drp-black: #FFFFFF;
  /* ... all tokens swap */
}
